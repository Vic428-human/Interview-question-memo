```
function calcManaRegen(intValue, spValue, recoveryRatePercent) {
  const recoveryMultiplier = 1 + recoveryRatePercent / 100;

  const baseRegen = Math.floor(1 + intValue / 6 + spValue / 100);
  const finalRegen = Math.floor(baseRegen * recoveryMultiplier);

  return finalRegen;
}

// 測試數值
const baseINT = 36;
const baseSP = 127;
const recoveryRatePercent = 37; // 玩家輸入的自然回復率

const regen = calcManaRegen(baseINT, baseSP, recoveryRatePercent);
console.log(regen); // Output: 11
```
